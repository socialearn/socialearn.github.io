10. nilai Tg 75 derajat + tg 15 derajat adalah


[tex]$\begin{align}\tan75+\tan15&=\tan(45+30)+\tan(45-30) \\ &=\frac{\tan 45+\tan 30}{1-\tan 45\tan 30}+\frac{\tan45-\tan 30}{1+\tan45\tan 30} \\ &=\frac{1+\frac{1}{3}\sqrt{3}}{1-1(\frac{1}{3}\sqrt{3})}+\frac{1-\frac{1}{3}\sqrt{3}}{1+1(\frac{1}{3}\sqrt{3})} \\ &=\frac{\frac{3+\sqrt{3}}{3}}{\frac{3-\sqrt{3}}{3}}+\frac{\frac{3-\sqrt{3}}{3}}{\frac{3+\sqrt{3}}{3}} \\ &=\frac{3+\sqrt{3}}{3-\sqrt{3}}+\frac{3-\sqrt{3}}{3+\sqrt{3}}\end{align}[/tex]

[tex]$\begin{align}\tan75+\tan 15&=\frac{(3+\sqrt{3})^2+(3-\sqrt{3})^2}{3^2-3} \\ &=\frac{9+6\sqrt{3}+3+(9-6\sqrt{3}+3)}{9-3} \\ &=\frac{24}{6} \\ &=4\end{align}[/tex]
